CREATE PROCEDURE [dbo].[SP_ListaTipoFormaPago]
AS
BEGIN
	SELECT Id, Nombre FROM TipoFormaPago WHERE Activo = 1
END

go

CREATE FUNCTION Fun_BuscarNombreTipoFormaPago (@idTipoFormaPago INT)
RETURNS VARCHAR(255)
AS BEGIN
DECLARE @NombreTipoFormaPago VARCHAR(255)
	SELECT	@NombreTipoFormaPago = Nombre
	FROM	TipoFormaPago
	WHERE	Id = @idTipoFormaPago
	RETURN @NombreTipoFormaPago
END
