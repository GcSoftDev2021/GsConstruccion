
CREATE TABLE TipoHoraExtra 
(Id int identity(1,1) Primary key ,
Nombre varchar(255),
Activo int default 1
unique(Nombre)
)

go

insert into TipoHoraExtra (nombre) values ('Hora Extra Diurna')
insert into TipoHoraExtra (nombre) values ('Hora Extra Nocturna')
insert into TipoHoraExtra (nombre) values ('Hora Extra Dominical y/o Festiva')

go

CREATE PROCEDURE [dbo].[SP_ListaTipoHoraExtra]
AS
BEGIN
	SELECT Id, Nombre FROM TipoHoraExtra WHERE Activo = 1
END

go

CREATE FUNCTION Fun_BuscarNombreTipoHoraExtra (@idTipoHoraExtra INT)
RETURNS VARCHAR(255)
AS BEGIN
DECLARE @NombreTipoHoraExtra VARCHAR(255)
	SELECT	@NombreTipoHoraExtra = Nombre
	FROM	TipoHoraExtra
	WHERE	Id = @idTipoHoraExtra 
	RETURN @NombreTipoHoraExtra
END
