
CREATE TABLE TipoEstadoCivil 
(Id int identity(1,1) Primary key ,
Nombre varchar(255),
Activo int default 1
unique(Nombre)
)

go

insert into TipoEstadoCivil (nombre) values ('Soltero(a)')

insert into TipoEstadoCivil (nombre) values ('Unión Libre')
insert into TipoEstadoCivil (nombre) values ('Viudo(a)')
insert into TipoEstadoCivil (nombre) values ('Separado(a)')

go

CREATE PROCEDURE [dbo].[SP_ListaTipoEstadoCivil]
AS
BEGIN
	SELECT Id, Nombre FROM TipoEstadoCivil WHERE Activo = 1
END

go

CREATE FUNCTION Fun_BuscarNombreTipoEstadoCivil (@idTipoEstadoCivil INT)
RETURNS VARCHAR(255)
AS BEGIN
DECLARE @NombreTipoEstadoCivil VARCHAR(255)
	SELECT	@NombreTipoEstadoCivil = Nombre
	FROM	TipoEstadoCivil
	WHERE	Id = @idTipoEstadoCivil
	RETURN @NombreTipoEstadoCivil
END
