
CREATE TABLE TipoDocumentoIdentificacion 
(Id int identity(1,1) Primary key ,
Nombre varchar(255),
Descripcion varchar(255),
Activo int default 1
unique(Nombre)
)

go

insert into TipoDocumentoIdentificacion (nombre, Descripcion) values ('CE', 'Cedula Extranjería')

go

CREATE PROCEDURE [dbo].[SP_ListaTipoDocumentoIdentificacion]
AS
BEGIN
	SELECT Id, Nombre FROM TipoDocumentoIdentificacion WHERE Activo = 1
END

go

CREATE FUNCTION Fun_BuscarNombreTipoDocumentoIdentificacion (@idTipoDocumentoIdentificacion INT)
RETURNS VARCHAR(255)
AS BEGIN
DECLARE @NombreTipoDocumentoIdentificacion VARCHAR(255)
	SELECT	@NombreTipoDocumentoIdentificacion = Nombre
	FROM	TipoDocumentoIdentificacion
	WHERE	Id = @idTipoDocumentoIdentificacion
	RETURN @NombreTipoDocumentoIdentificacion
END
